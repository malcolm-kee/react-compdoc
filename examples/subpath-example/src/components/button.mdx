## `useState` Example

```tsx
const Demo = () => {
  const [on, setOn] = React.useState(false);

  return <Button onClick={() => setOn(!on)}>{on ? 'On' : 'Off'}</Button>;
};

<Demo />;
```

## Example that fetch data

```jsx
const Example = () => {
  const [result, setResult] = React.useState('');

  return (
    <div>
      <Button
        onClick={() =>
          fetch('/hello')
            .then((res) => res.json())
            .then((data) => setResult(data.message))
        }
        variant="primary"
      >
        Fetch Data
      </Button>
      <div>
        Result:
        <output>{result}</output>
      </div>
    </div>
  );
};

render(<Example />);
```
